### 1. kind 클러스터 생성 및 삭제
# 1) 클러스터 생성
kind create cluster --config ./kind-config.yaml 

# 2) 클러스터 삭제
kind delete cluster


### 2. 대시보드 설치
# 1) 대시보드 설치하기
kubectl apply -f https://raw.githubusercontent.com/kubernetes/dashboard/v2.7.0/aio/deploy/recommended.yaml

# [이슈] 대시보드 설치 시 TLS handshake timeout 에러 발생
# [원인] 호스트/경로로 가는 HTTPS가 중간에서 막히거나 끊기는 네트워크 문제
# [해결] 다른 네트워크에서 시도하거나 파일을 로컬로 받은 뒤 적용
kubectl apply -f dashboard.yaml

# 2) 접속을 위한 프록시 실행
# [주의] 이 명령어를 실행하면 터미널이 대기 상태가 됩니다. 대시보드를 쓰는 동안 이 터미널은 그대로 두시고, 다른 작업을 하려면 새 터미널 탭을 여세요.
kubectl proxy

# 3) 접속 주소
http://localhost:8001/api/v1/namespaces/kubernetes-dashboard/services/https:kubernetes-dashboard:/proxy/

# 4) 로그인 토큰 생성하기 (가장 중요)
# 4.1) 서비스 어카운트 생성
kubectl create serviceaccount admin-user -n kubernetes-dashboard

# 4.2) 관리자 권한 부여
kubectl create clusterrolebinding admin-user --clusterrole=cluster-admin --serviceaccount=kubernetes-dashboard:admin-user

# 4.3) 토큰 확인
kubectl -n kubernetes-dashboard create token admin-user



### 3. Ingress Controller 설치 (kind 기준)
# 1) 클러스터에 아무 Ingress Controller도 없으면 Ingress 리소스만 켜도 트래픽이 안 들어옵니다. kind에서는 보통 ingress-nginx를 씁니다.
# 1.1) [오류] 네트워크에서 막는 경우가 있음
kubectl apply -f https://raw.githubusercontent.com/kubernetes/ingress-nginx/main/deploy/static/provider/kind/deploy.yaml
# 1.2) [해결] 직접 설치 : 개인 핫스팟과 같이 네트워크 변경 후 설치
curl -L -o ingress-nginx-kind.yaml https://raw.githubusercontent.com/kubernetes/ingress-nginx/main/deploy/static/provider/kind/deploy.yaml
kubectl apply -f ingress-nginx-kind.yaml


# 2) 설치 후 컨트롤러가 뜰 때까지:
kubectl wait --namespace ingress-nginx \
  --for=condition=ready pod \
  --selector=app.kubernetes.io/component=controller \
  --timeout=120s

# 3) sample 차트에서 Ingress 켜기
sample/values.yaml에서 아래처럼 바꿉니다.

ingress:
  enabled: true
  className: nginx
  annotations: {}
  hosts:
    - host: sample.local
      paths:
        - path: /api/departments
          pathType: Prefix
          component: backend1
        - path: /api/employees
          pathType: Prefix
          component: backend2
        - path: /
          pathType: Prefix
          component: frontend

# 4) 브라우저에서 주소 쓰는 방법
# 4.1) 방법 A: 포트 포워딩 (kind 설정 안 바꿀 때)
kubectl port-forward -n ingress-nginx service/ingress-nginx-controller 8888:80

# 4.2) 방법 B: kind에 80 포트 노출 (클러스터 재생성 필요)
kind-config.yaml의 control-plane 노드에 extraPortMappings로 호스트 80 → 컨테이너 80을 열어 두면, kind 문서처럼 http://sample.local 만으로 접속할 수 있습니다. 클러스터를 다시 만들어야 해서, 이미 쓰는 클러스터가 있으면 방법 A가 부담이 적습니다.
